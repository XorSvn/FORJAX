#

#    chmod +x install.sh

#  descarga todo lo necesario automaticamente.

#  ┌─────────────────────────────────────────────────┐

pyinstaller>=6.0

# ──────────────────────────────────────────────────
#  [PYTHON] — ya cubierto con pyinstaller arriba
#  pip install pyinstaller
#
#  [BASH]
#    Windows : instala Git Bash → https://git-scm.com
#              o activa WSL     → https://aka.ms/wsl
#    Linux   : sudo apt install shc
#    Mac     : brew install shc
#
#  [C]
#    Windows : instala MinGW-w64 → https://winlibs.com
#    Linux   : sudo apt install gcc build-essential
#    Mac     : brew install gcc
#
#  [GO]
#    Windows / Linux / Mac → https://go.dev/dl/
#    Linux   : sudo apt install golang-go
#    Mac     : brew install go
#
#  [JAVA]
#    JDK (requerido):
#      Windows / Linux / Mac → https://adoptium.net
#      Linux   : sudo apt install default-jdk
#      Mac     : brew install openjdk
#    Launch4j (opcional, para generar .exe desde .jar):
#      https://launch4j.sourceforge.net
#
#  [JAVASCRIPT]
#    Node.js (requerido):
#      Windows / Linux / Mac → https://nodejs.org
#      Linux   : sudo apt install nodejs npm
#      Mac     : brew install node
#    pkg (empaquetador a .exe):
#      npm install -g pkg
#
#  [PHP]
#    Windows : https://www.php.net/downloads
#    Linux   : sudo apt install php-cli
#    Mac     : brew install php
#    IMPORTANTE: asegurate de tener phar.readonly=Off
#                en tu archivo php.ini
#
#  [POWERSHELL] — solo Windows
#    Ejecuta esto en PowerShell como Administrador:
#      Install-Module ps2exe -Scope CurrentUser -Force
#
